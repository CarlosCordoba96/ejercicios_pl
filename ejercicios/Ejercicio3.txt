grammar Ejercicio3{
	analysis LL1;
	terminal a,b,c;
	nonterminal E, E1;
	
	E := a E1 | b E1 | c E1 | '(' E ')' E1;
	E1 := '+' E E1 | '*' E E1 | ;
}

solution Ejercicio3{
	nullables{
		E1;
	}
	firsts{
		E = {a, b, c, '('};
		E1 = {'+', '*'};
	}
	followers{
		E = {'+', '*', ')', $};
		E1 = {'+', '*', ')', $};
	}
	lookaheads{
		1 = {a};
		2 = {b};
		3 = {c};
		4 = {'('};
		5 = {'+'};
		6 = {'*'};
		7 = {'+', '*', ')', $};
	}
}
